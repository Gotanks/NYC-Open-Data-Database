(COMMITTEE_ID CHAR(9) ,

 FEC_ELECTION_TYPE_DESC VARCHAR(100),
 FEC_ELECTION_YEAR VARCHAR(10),

CREATE TABLE CONTRIBUTOR
(CONTRIBUTOR_NAME VARCHAR(100) NOT NULL,
 TRANSATION_ID VARCHAR PRIMARY KEY NOT NULL,
 CONTRIBUTOR_FIRST_NAME VARCHAR(20),
 COTRIBUTOR_MIDDLE_NAME VARCHAR(20),
 CONTRIBUTOR_LAST_NAME VARCHAR(50),
 CONTRIBUTOR_SUFFIX VARCHAR(20)
 CONTRIBUTOR_STREET_1 VARCHAR(100),
 CONTRIBUTOR_STREET_2 VARCHAR(100),
 CONTRIBUTION_RECEIPT_DATE TEXT,
 CONTRIBUTION_RECEIPT_AMOUNT INTEGER,
 CONTRIBUTOR_AGGREGATE_YTD INTEGER,
 RECEIPT_TYPE VARCHAR(20),
 );
CREATE TABLE CONTRIBUTOR_OCCUPATION
